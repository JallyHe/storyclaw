import fs from 'fs/promises'
import path from 'path'
import { defineTool } from '@earendil-works/pi-coding-agent'
import { Type } from '@sinclair/typebox'
import { buildTree, readStoryFile, writeStoryFile } from '../fs/workspace'
import { parseFile, serializeFile } from '../fs/serializer'
import type { BrowserWindow } from 'electron'
import type { StoryFile, DiffBlock, Block } from '../../src/types'

let workspaceRoot = ''
let currentWin: BrowserWindow | null = null

export function initTools(root: string, win: BrowserWindow) {
  workspaceRoot = root
  currentWin = win
}

// ── block-level diff ──────────────────────────────────────────────────────────
function computeBlockDiff(oldFile: StoryFile, newFile: StoryFile): DiffBlock[] {
  if (!('blocks' in oldFile) || !('blocks' in newFile)) return []
  const oldBlocks = (oldFile as any).blocks as Block[]
  const newBlocks = (newFile as any).blocks as Block[]
  const oldIds = new Set(oldBlocks.map(b => b.id))
  const newIds = new Set(newBlocks.map(b => b.id))

  const result: DiffBlock[] = []
  // kept and added
  for (const blk of newBlocks) {
    result.push({ blk, diff: oldIds.has(blk.id) ? null : 'add' })
  }
  // deleted (insert after nearest preceding kept block)
  for (const blk of oldBlocks) {
    if (!newIds.has(blk.id)) result.push({ blk, diff: 'del' })
  }
  return result
}

// ── tools ──────────────────────────────────────────────────────────────────────

export const readScreenplay = defineTool({
  name: 'read_screenplay',
  label: '读取剧本文件',
  description: '读取工作区内的剧本文件（.ep / .chr / .otl / .wld），返回 JSON 内容',
  parameters: Type.Object({
    path: Type.String({ description: '相对于工作区根目录的文件路径，例如 剧集/EP01 幽灵来电.ep' })
  }),
  execute: async (_id, { path: relPath }) => {
    const abs = path.resolve(workspaceRoot, relPath)
    const data = await readStoryFile(abs)
    return { content: [{ type: 'text' as const, text: JSON.stringify(data, null, 2) }], details: {} }
  }
})

export const writeScreenplay = defineTool({
  name: 'write_screenplay',
  label: '修改剧本文件',
  description: '将修改后的完整 JSON 内容写入剧本文件。改动不会立即写入磁盘，而是以 diff 形式发送给用户审核。',
  parameters: Type.Object({
    path: Type.String({ description: '相对于工作区根目录的文件路径' }),
    content: Type.String({ description: '完整的 JSON 字符串（必须符合对应文件格式）' })
  }),
  execute: async (_id, { path: relPath, content }) => {
    const abs = path.resolve(workspaceRoot, relPath)
    const ext = path.extname(abs).slice(1).toLowerCase()
    const newFile = parseFile(ext as any, content)
    const currentRaw = await fs.readFile(abs, 'utf-8').catch(() => '{"version":1,"blocks":[]}')
    const currentFile = parseFile(ext as any, currentRaw)

    const diffBlocks = computeBlockDiff(currentFile, newFile)
    currentWin?.webContents.send('agent:event', {
      type: 'change',
      fileId: abs,
      diffBlocks,
      newContent: newFile
    })

    return {
      content: [{ type: 'text' as const, text: `diff 已发送给用户审核：${relPath}（${diffBlocks.filter(d => d.diff === 'add').length} 新增，${diffBlocks.filter(d => d.diff === 'del').length} 删除）` }],
      details: {}
    }
  }
})

export const listWorkspace = defineTool({
  name: 'list_workspace',
  label: '列出工作区',
  description: '列出工作区的完整文件树（文件夹和文件）',
  parameters: Type.Object({}),
  execute: async () => {
    const tree = await buildTree(workspaceRoot)

export const readReference = defineTool({
  name: 'read_reference',
  label: '读取参考资料',
  description: '读取参考/目录下的文档（txt / md / pdf），用于小说改编、参考资料分析等',
  parameters: Type.Object({
    path: Type.String({ description: '相对于工作区根目录的文件路径，例如 参考/原著小说.txt' })
  }),
  execute: async (_id, { path: relPath }) => {
    const abs = path.resolve(workspaceRoot, relPath)
    const ext = path.extname(abs).slice(1).toLowerCase()
    let text: string
    if (ext === 'pdf') {
      const pdfParse = (await import('pdf-parse')).default
      const buf = await fs.readFile(abs)
      const result = await pdfParse(buf)
      text = result.text
    } else {
      text = await fs.readFile(abs, 'utf-8')
    }
    // truncate to 8000 chars to fit context
    if (text.length > 8000) text = text.slice(0, 8000) + '\n…（内容已截断）'
    return { content: [{ type: 'text' as const, text }], details: {} }
  }
})

export const ALL_TOOLS = [readScreenplay, writeScreenplay, listWorkspace, readReference]

- [ ] **Step 2: Create `electron/agent/streaming.ts`**

```typescript
import type { BrowserWindow } from 'electron'

const TOOL_LABELS: Record<string, string> = {
  read_screenplay:  '读取剧本文件',
  write_screenplay: '修改剧本文件',
  list_workspace:   '列出工作区',
  read_reference:   '读取参考资料'
}

export function setupStreaming(session: any, win: BrowserWindow) {
  // accumulate text delta into a buffer for current reply line
  let textBuffer = ''

  session.subscribe((event: any) => {
    switch (event.type) {
      case 'message_update': {
        const ae = event.assistantMessageEvent
        if (ae?.type === 'text_delta') {
          textBuffer += ae.delta
          win.webContents.send('agent:event', { type: 'text_delta', delta: ae.delta })
        }
        break
      }
      case 'tool_execution_start':
        win.webContents.send('agent:event', {
          type: 'tool_start',
          tool: event.toolName,
          label: TOOL_LABELS[event.toolName] ?? event.toolName,
          target: event.toolInput?.path ?? ''
        })
        break
      case 'tool_execution_end':
        win.webContents.send('agent:event', {
          type: 'tool_end',
          tool: event.toolName,
          isError: !!event.isError
        })
        break
      case 'agent_end':
        textBuffer = ''
        win.webContents.send('agent:event', { type: 'agent_end' })
        break
    }
  })
}
```

- [ ] **Step 1: Create `src/hooks/useAgentEvents.ts`**

```typescript
import { useEffect } from 'react'
import { useSessionsStore } from '@/store'
import { useChangesStore } from '@/store'
import { agentIpc } from '@/ipc/agent'
import type { AgentEvent } from '@/types'

export function useAgentEvents() {
  const { addMessage, appendDelta, addToolStep, completeToolStep, finalizeReply } = useSessionsStore()
  const { addChange } = useChangesStore()

  useEffect(() => {
    const unsub = agentIpc.onEvent((event: AgentEvent) => {
      switch (event.type) {
        case 'text_delta':
          appendDelta(event.delta)
          break
        case 'tool_start':
          addToolStep(event.tool, event.label, event.target)
          break
        case 'tool_end':
          completeToolStep(event.tool, event.isError)
          break
        case 'change':
          addChange({ fileId: event.fileId, diffBlocks: event.diffBlocks, newContent: event.newContent })
          break
        case 'agent_end':
          finalizeReply()
          break
      }
    })
    return unsub
  }, [appendDelta, addToolStep, completeToolStep, finalizeReply, addChange])
}

import { useState, useRef, useCallback, useEffect } from 'react'
import { agentIpc } from '@/ipc/agent'
import { useSessionsStore } from '@/store'
import type { AgentMode } from '@/types'

const MODES: { id: AgentMode; label: string }[] = [
  { id: 'craft', label: 'Craft' },
  { id: 'plan',  label: 'Plan' },
  { id: 'ask',   label: 'Ask' }
]

interface Props {
  busy: boolean
  onBusyChange: (busy: boolean) => void
  placeholder?: string
  big?: boolean
}

export function AgentComposer({ busy, onBusyChange, placeholder = '向 AI 提问或描述写作任务…', big }: Props) {
  const [text, setText] = useState('')
  const [mode, setMode] = useState<AgentMode>('craft')
  const { addMessage, sessions, activeId } = useSessionsStore()
  const ref = useRef<HTMLTextAreaElement>(null)

  // auto-resize textarea
  useEffect(() => {
    const el = ref.current
    if (!el) return
    el.style.height = 'auto'
    el.style.height = Math.min(el.scrollHeight, big ? 200 : 120) + 'px'
  }, [text, big])

  const send = useCallback(async () => {
    const t = text.trim()
    if (!t || busy) return
    setText('')
    addMessage({ role: 'user', text: t })
    addMessage({ role: 'assistant', steps: [], reply: [], typing: true })
    onBusyChange(true)
    await agentIpc.send(t, mode)
  }, [text, busy, mode, addMessage, onBusyChange])

  const stop = useCallback(async () => {
    await agentIpc.stop()
    onBusyChange(false)
  }, [onBusyChange])

  const onKey = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); send() }
  }

  return (
    <div className="composer">
      <div className="composer-input-row">
        <textarea
          ref={ref}
          className="composer-textarea"
          rows={big ? 3 : 1}
          value={text}
          onChange={e => setText(e.target.value)}
          onKeyDown={onKey}
          placeholder={placeholder}
          disabled={busy}
        />
        {busy
          ? <button className="composer-stop" onClick={stop} title="停止">■</button>
          : <button className="composer-send" disabled={!text.trim()} onClick={send} title="发送">↑</button>
        }
      </div>
      <div className="composer-meta">
        <select className="meta-select" value={mode} onChange={e => setMode(e.target.value as AgentMode)}>
          {MODES.map(m => <option key={m.id} value={m.id}>{m.label}</option>)}
        </select>
      </div>
    </div>
  )
}
```

import { useSessionsStore } from '@/store'
import './agent.css'

interface Props { width: number }

export function SessionList({ width }: Props) {
  const { sessions, activeId, setActive, newSession } = useSessionsStore()

  const groups = sessions.reduce<Record<string, typeof sessions>>((acc, s) => {
    if (!acc[s.group]) acc[s.group] = []
    acc[s.group].push(s)
    return acc
  }, {})

  return (
    <div className="session-list" style={{ width, flexShrink: 0 }}>
      <div className="session-list-head">
        <span>会话</span>
        <button className="session-new-btn" onClick={newSession}>+ 新建</button>
      </div>
      <div className="session-scroll">
        {Object.entries(groups).map(([group, items]) => (
          <div key={group}>
            <div className="session-group-label">{group}</div>

              <div key={s.id} className={`session-item${s.id === activeId ? ' active' : ''}`} onClick={() => setActive(s.id)}>
                <div className="session-item-title">{s.title}</div>
                <div className="session-item-time">{s.time}</div>
              </div>
            ))}
          </div>
        ))}
      </div>
    </div>
  )
}

function DiffPreview({ blocks }: { blocks: DiffBlock[] }) {
  const visible = blocks.filter(b => b.diff !== null).slice(0, 12)
  return (
    <div className="change-diff-preview">
      {visible.map((item, i) => {
        const text = 'text' in item.blk ? item.blk.text : item.blk.type === 'scene' ? `[场${item.blk.number} ${item.blk.location}]` : `[${item.blk.type}]`
        return (
          <span key={i} className={item.diff === 'add' ? 'cdiff-add' : 'cdiff-del'}>
            {item.diff === 'add' ? '+ ' : '− '}{text}
          </span>
        )
      })}
    </div>
  )
}

interface Props { width: number }

export function ChangesPanel({ width }: Props) {
  const { changes, acceptChange, rejectChange, acceptAll, rejectAll } = useChangesStore()
  const { openTab } = useTabsStore()
  const hasChanges = changes.size > 0

  return (
    <div className="changes-panel" style={{ width, flexShrink: 0 }}>
      <div className="changes-head">
        <span>改动 {hasChanges ? `(${changes.size})` : ''}</span>
        {hasChanges && (
          <div className="changes-all-btns">
            <button className="changes-all-btn btn-accept-all" onClick={acceptAll}>全部接受</button>
            <button className="changes-all-btn btn-reject-all" onClick={rejectAll}>全部拒绝</button>
          </div>
        )}
      </div>
      <div className="changes-scroll">
        {!hasChanges
          ? <div className="changes-empty">暂无待审改动</div>
          : [...changes.entries()].map(([fileId, change]) => {
              const name = path.basename(fileId)
              const adds = change.diffBlocks.filter(b => b.diff === 'add').length
              const dels = change.diffBlocks.filter(b => b.diff === 'del').length
              return (
                <div key={fileId} className="change-item">
                  <div className="change-item-head">
                    <span className="change-item-name" onClick={() => openTab(fileId)} style={{ cursor: 'pointer' }}>{name}</span>
                    {adds > 0 && <span className="change-stat-add">+{adds}</span>}
                    {dels > 0 && <span className="change-stat-del">−{dels}</span>}
                    <div className="change-item-btns">
                      <button className="ch-btn-accept" onClick={() => acceptChange(fileId)}>接受</button>
                      <button className="ch-btn-reject" onClick={() => rejectChange(fileId)}>拒绝</button>
                    </div>
                  </div>
                  <DiffPreview blocks={change.diffBlocks} />
                </div>
              )
            })
        }
      </div>
    </div>
  )
}
